Estructura Meta Texto Enlaces

Estructura

Plantilla recomendada

                    
                    <!-- Ejemplo: estructura mínima --> 
                        <!DOCTYPE html> 
                            <html lang="es"> 
                                <head> 
                                    <meta charset="UTF-8"> 
                                    <meta name="viewport" content="width=device-width, initial-scale=1.0"> 
                                     <title>Título</title> 
                                </head> 
                                <body> 
                                    <main>Contenido</main> 
                                </body> 
                        </html>

Meta

Básicos + seo + redes

                    
                        <!-- Configuración de caracteres y vista móvil -->
                        <meta charset="UTF-8">
                        <meta name="viewport" content="width=device-width, initial-scale=1.0">
                        <meta name="description" content="Descripción clara">

                        <!-- SEO: Indexación y URL original -->
                        <meta name="robots" content="index, follow">
                        <link rel="canonical" href="https://ejemplo.com/pagina">

                        <!-- Open Graph: Así se verá en Facebook/WhatsApp -->
                        <meta property="og:title" content="Título">
                        <meta property="og:description" content="Descripción">
                        <meta property="og:image" content="https://ejemplo.com/img.jpg">

Texto

Jerarquia de titulos

                            
                        <!-- Jerarquía de encabezados -->
                        <h1>Título principal</h1>
                        <h2>Sección</h2>
                        <p>Párrafo normal.</p>

                        <!-- Estilos de énfasis y resaltado -->
                        <strong>Importante</strong>
                        <em>Énfasis</em>
                        <mark>Highlight</mark>

                        <!-- Código y bloques -->
                        <code>console.log("inline")</code>
                        <pre>Bloque preformateado</pre>

Enlaces

                        
                        <!-- Tipos de enlaces -->
                        <a href="https://ejemplo.com">Externo</a>
                        <a href="/ruta">Interno</a>
                        <a href="#seccion">Ancla</a>

                        <!-- Enlace a nueva pestaña con seguridad -->
                        <a href="https://ejemplo.com" target="_blank" rel="noopener noreferrer">
                        Nueva pestaña (seguro)
                        </a>

                        <!-- Estructura de navegación accesible -->
                        <nav aria-label="Principal">
                        <ul>
                            <li><a href="#inicio">Inicio</a></li>
                        </ul>
                        </nav>
